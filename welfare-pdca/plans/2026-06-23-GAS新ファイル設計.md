#### 暦日 LEFT JOIN SQL（中核ロジック）

```sql
WITH calendar AS (
  SELECT generate_series(
    DATE :start_date,
    DATE :end_date,
    INTERVAL '1 day'
  )::DATE AS "記録日"
),
users AS (
  SELECT DISTINCT "利用者在籍ID"
  FROM public."児童入所登録"   -- ← category で動的選択（児童入所登録 / 短期入所登録 / 日中一時登録）
  WHERE "年月" = :nengetsu
    AND "利用者在籍ID" = ANY(:user_ids)
),
matrix AS (
  SELECT u."利用者在籍ID", c."記録日"
  FROM users u CROSS JOIN calendar c
)
SELECT
  m."利用者在籍ID",
  m."記録日",
  TO_CHAR(m."記録日", 'DD')  AS "日",
  r."利用状態",          -- 入所系のみ
  r."利用類型",          -- 短期入所のみ
  r."時間区分",          -- 日中一時のみ
  r."基本報酬",
  r."加算",
  r."区分選択肢",
  r."実費1", r."実費2", r."実費3", r."実費4", r."実費5"
FROM matrix m
LEFT JOIN public."児童入所登録" r
  ON r."利用者在籍ID" = m."利用者在籍ID"
  AND r."記録日" = m."記録日"
  AND r."フラグ" IS DISTINCT FROM TRUE
ORDER BY m."利用者在籍ID", m."記録日";
```

#### 反映条件の処理（中核ロジック）

```javascript
// AppSheet DB 「001_利用者加算マスタ」を Application API v2 で取得
function loadKasanMaster(category) {
  const accessKey = PropertiesService.getScriptProperties().getProperty('APPSHEET_API_KEY_主app');
  const url = `https://www.appsheet.com/api/v2/apps/${appId}/tables/001_利用者加算マスタ/Action`;
  const body = {
    Action: "Find",
    Properties: { Locale: "ja-JP" },
    Rows: [{}]
  };
  const res = UrlFetchApp.fetch(url, {
    method: 'post', contentType: 'application/json',
    headers: { 'ApplicationAccessKey': accessKey },
    payload: JSON.stringify(body)
  });
  const all = JSON.parse(res.getContentText());
  // category と同じ事業所REF のみフィルタ
  return all.filter(r => r['事業所REF'] === category)
            .reduce((m, r) => {
              m[r['利用者加算一覧']] = {
                kubun: r['区分_選択肢'],
                kind: r['加算種別'],
                reflect_cond: r['反映条件'],
                reflect_where: r['反映箇所']
              };
              return m;
            }, {});
}

// 加算集計
function aggregateKasan(rows, kasanMaster) {
  const userMap = {}; // uID → { dailies: [], kasanMap: {}, kasanCountMap: {} }
  rows.forEach(row => {
    const uID = row['利用者在籍ID'];
    if (!userMap[uID]) userMap[uID] = { dailies: [], kasanMap: {}, kasanCountMap: {} };

    // 日次行展開（暦日固定）
    userMap[uID].dailies.push({
      day: row['日'],
      jotai: row['利用状態'],
      kihon: row['基本報酬'] || '',
      kasan: row['加算'] || '',
      cost1: row['実費1'], cost2: row['実費2'], cost3: row['実費3'],
      cost4: row['実費4'], cost5: row['実費5']
    });

    // 加算集計（反映条件別）
    const kasanStr = row['加算'] || '';
    kasanStr.split(',').filter(k => k.trim()).forEach(k => {
      const kasan = k.trim();
      const meta = kasanMaster[kasan];
      if (!meta) return; // 未定義加算はスキップ

      if (meta.reflect_cond === '同月全件反映_カウント') {
        // 日数カウント
        if (!userMap[uID].kasanCountMap[kasan]) {
          userMap[uID].kasanCountMap[kasan] = { count: 0, kubun: row['区分選択肢'] };
        }
        userMap[uID].kasanCountMap[kasan].count++;
      } else {
        // 同月最新日のみ反映（既存ロジック）
        userMap[uID].kasanMap[kasan] = row['区分選択肢'];
      }
    });
  });
  return userMap;
}
```

#### Excel テンプレ転記ロジック

```javascript
// 暦日固定行への展開（28〜31 行を常に出力）
function writeDailyRowsFixed(targetSheet, user, daysInMonth, dRow) {
  // dRow: テンプレ上の「日」行のインデックス
  // user.dailies は暦日 LEFT JOIN 済なので 28〜31 件すべてある
  const arrA = [], arrB = [], arrC = [], arrE = [], arrF = [];
  for (let i = 0; i < daysInMonth; i++) {
    const d = user.dailies[i] || { day: String(i + 1).padStart(2, '0'), kihon: '', cost1: '', cost2: '' };
    arrA.push([d.day]);
    arrB.push([`=IF(A${dRow + i}<>"",TEXT(DATE(LEFT($B$3,4),RIGHT($B$3,2),A${dRow + i}),"aaa"),"")`]);
    arrC.push([d.kihon || '']);
    arrE.push([d.cost1 || '']);
    arrF.push([d.cost2 || '']);
  }
  targetSheet.getRange(dRow, 1, daysInMonth, 1).setValues(arrA);
  targetSheet.getRange(dRow, 2, daysInMonth, 1).setFormulas(arrB);  // ★ 曜日数式
  targetSheet.getRange(dRow, 3, daysInMonth, 1).setValues(arrC);
  targetSheet.getRange(dRow, 5, daysInMonth, 1).setValues(arrE);
  targetSheet.getRange(dRow, 6, daysInMonth, 1).setValues(arrF);
}

// 加算集計値の転記
function writeKasanSummary(targetSheet, user) {
  // user.kasanMap (同月最新日) と user.kasanCountMap (同月全件カウント) を統合
  const allKasan = { ...user.kasanMap, ...user.kasanCountMap };
  Object.entries(allKasan).forEach(([kasan, value]) => {
    // テンプレ A 列で加算名を検索 → B 列に値を書込
    const targetRow = findRow(targetSheet, kasan);
    if (targetRow > 0) {
      // 同月全件反映_カウント の場合は日数（count）を書く
      // 同月最新日のみ反映 の場合は区分（kubun）を書く
      const cellValue = typeof value === 'object' ? value.count : value;
      targetSheet.getRange(targetRow, 2).setValue(cellValue);
    }
  });
}
```

## §3. AppSheet Bot 側の設定変更

請求アプリの **Bot:レセBOT** で **Process（Call a script）の FunctionName を category 別に分岐** させる：

```yaml
レセBOT トリガ → Branch by category:
  IF [category] IN ("児童入所施設", "短期入所", "日中一時支援"):
    → Call a script: makeReceptInsho
  ELIF [category] IN ("計画相談支援", "障害児相談支援") AND [年月] >= "2026-06":
    → Call a script: makeReceptSoudan6Kaitei
  ELSE:
    → Call a script: makeRecept (既存・温存)
```

または、エントリ関数 `makeRecept` を**ルータ化**：

## §7. リスク・懸念

| リスク | 対策 |
|---|---|
| 暦日 LEFT JOIN の SQL コストが大きい | 月単位なので 31 行 × 利用者数 で限定的。INDEX 設計済 |
| 加算マスタ取得 API のレイテンシ | makeRecept の冒頭で 1 回のみ取得、メモ化 |
| 反映条件未定義の加算 | デフォルト「同月最新日のみ反映」で fallback |
| ユニットテスト不在 | Test 1〜4 を実装、CI なしの環境では手動チェック |
| 入所系の入所〜退所をまたぐ加算 | 利用状態列で「在籍」と「退所中」を区別、退所中は加算対象外 |
